   let fastest = workers |> List.maxBy (fun w -> w.CpuCount * w.CpuClock)
   let! task = Cloud.StartChild(computation, target = fastest)
   return task
}

var path = "http://dumps.wikimedia.org/other/static_html_dumps/current/en/html.lst";

CloudFlow.OfHttpFileByLine(path)
    .SelectMany(line => line.Split(' '))
    .Select(word => word.Trim())
    .Where(word => word.Length > 3)
    .CountBy(word => word)
    .OrderByDescending(count => count.Value)
    .ToArray()
