let getLengthAsync (uri : Uri) = async {
    use wc = new WebClient()
    let! lines = wc.AsyncDownloadString uri
    return lines.Split('\n').Length
}

    let! sizes = Async.Parallel [for u in uris -> getLengthAsync u]
    return Array.sum sizes
