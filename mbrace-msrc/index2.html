type AsyncBuilder() =
    member __.Return(t:'T) : Async<'T> = fun _ k -> k t
    member __.Bind(f:Async<'T>, g:'T -> Async<'S>) : Async<'S> =
        fun c k -> 
            if c.IsCancellationRequested then () 
            else f c (fun t -> g t c k)

    fun c k ->
      let n,results = ref 0, Array.zeroCreate<'T> xs.Length
      let runChild (i:int) (ch:Async<'T>) =
        ch c (fun t ->
          results.[i] <- t
          if Interlocked.Increment n = xs.Length then k results)

      for i = 0 to xs.Length - 1 do
        ThreadPool.QueueUserWorkItem(fun () -> runChild i xs.[i])

type Cloud with
  static member Parallel(xs:Cloud<'T>[]) : Cloud<'T[]> =
    fun resources k ->
      let sched = resources.Resolve<IScheduler>()
      let n,results = ref 0, Array.zeroCreate<'T> xs.Length
      let runChild (i:int) (ch:Cloud<'T>) =
        ch resources (fun t ->
          results.[i] <- t
          if Interlocked.Increment n = xs.Length then k results)

      for i = 0 to xs.Length - 1 do
        sched.Enqueue(fun () -> runChild i xs.[i])
